Not every game object exists from the start. Enemies spawn as you progress, projectiles are created on each shot, pickups appear when a chest opens. Fraxel's `container.node.spawn()` method lets you add new nodes to the scene at runtime from any JSX component, giving you full control over when objects are created and destroyed.

## Why Dynamic Spawning

The JSX tree you write in a component is rendered once when the node starts. Spawning is how you add children after that initial render — programmatically, in response to events or signals — without re-rendering the entire tree.

## `container.node.spawn()`

The `spawn()` method on a container node takes any JSX element and adds it as a child of that container at runtime:

```tsx Spawner.tsx
import { useTransform, useClickable, useTrigger } from 'fraxel/hooks'

function Spawner() {
  const container = useTransform()
  const btn = useClickable()

  useTrigger(btn.clicked, () => {
    container.node.spawn(<Enemy position={[Math.random() * 400, 0]} />)
  })

  return (
    <transform ref={container}>
      <clickable ref={btn} size={[64, 32]}>
        <sprite textureId={SPAWN_BTN} />
      </clickable>
    </transform>
  )
}
```

The spawned node is rendered and started immediately. It behaves identically to a node declared statically in JSX.

<Note>
  Always pass a ref to the **container** node, not to the children you want to spawn. The container is the parent that newly spawned nodes will be attached to.
</Note>

## Destroying Nodes

Call `.destroy()` on any node to remove it from the scene and clean up all its listeners:

```tsx bullet.tsx
import { useRigidBody, useCollider, useTrigger } from 'fraxel/hooks'
import { shapes } from 'fraxel'

function Bullet({ position }: { position: [number, number] }) {
  const body = useRigidBody()
  const collider = useCollider()

  useTrigger(collider.colliderEntered, () => {
    // Remove this bullet from the scene
    body.node.destroy()
  })

  return (
    <rigid-body ref={body} position={position} useGravity={false}>
      <sprite textureId={BULLET} sourceSize={[8, 4]} />
      <collider
        ref={collider}
        shape={shapes.circle(4)}
        group={['bullet']}
        collidesWith={['enemy', 'wall']}
      />
    </rigid-body>
  )
}
```

## Cross-Component Communication with `useTrigger`

When two components need to communicate — for example, a child component tells a parent to spawn something — avoid prop drilling. Use `createTrigger` and `useTrigger` instead:

```tsx triggers.ts
import { createTrigger } from 'fraxel/hooks'

// Define triggers at module scope, shared between components
export const onFireBullet = createTrigger<[{ x: number; y: number }]>()
export const onEnemyDied = createTrigger<[]>()
```

```tsx gun.tsx
import { useAction, useEffect } from 'fraxel/hooks'
import { Input } from 'fraxel'
import { onFireBullet } from './triggers.js'

function Gun() {
  const fireAction = useAction(Input.createAction({ key: ' ' }))

  useEffect(() => {
    if (fireAction.justPressed()) {
      onFireBullet.emit({ x: 200, y: 150 })
    }
  })

  return <sprite textureId={GUN} />
}
```

```tsx game-scene.tsx
import { useTransform, useTrigger } from 'fraxel/hooks'
import { onFireBullet } from './triggers.js'

function GameScene() {
  const container = useTransform()

  useTrigger(onFireBullet, ({ x, y }) => {
    container.node.spawn(<Bullet position={[x, y]} />)
  })

  return (
    <transform ref={container}>
      <Player />
    </transform>
  )
}
```

`useTrigger` subscriptions are automatically cleaned up when the node is destroyed — no manual unsubscription needed.

## Reactive Arrays with `List`

When you have a signal that holds an array of data and you want to render a node for each item, use the `List` component from `fraxel/jsx`. It performs efficient keyed reconciliation — only adding and removing nodes when items are added or removed:

```tsx enemy-list.tsx
import { List } from 'fraxel/jsx'
import { useSignal } from 'fraxel/hooks'

interface EnemyData {
  id: number
  x: number
  y: number
}

function EnemyManager() {
  const [enemies, setEnemies] = useSignal<EnemyData[]>([])

  const addEnemy = () => {
    setEnemies([...enemies(), { id: Date.now(), x: Math.random() * 400, y: 0 }])
  }

  const removeEnemy = (id: number) => {
    setEnemies(enemies().filter((e) => e.id !== id))
  }

  return (
    <transform>
      <List array={enemies} itemKey={(e) => e.id}>
        {(enemy) => (
          <transform position={[enemy.x, enemy.y]}>
            <sprite textureId={ENEMY} />
          </transform>
        )}
      </List>
    </transform>
  )
}
```

`List` props:

| Prop | Description |
|------|-------------|
| `array` | `SignalGetter<T[]>` — the reactive data array |
| `itemKey` | `(item, index, arr) => string \| symbol` — unique key per item |
| `children` | `(item, index, arr) => Node` — render function for each item |
| `empty` | Optional fallback element when the array is empty |

## Complete Shooter Example

Here's a full shooter combining `spawn()`, `useTrigger`, and bullet destruction:

```tsx shooter-game.tsx
import { useTransform, useRigidBody, useCollider, useTrigger, useAction, useEffect } from 'fraxel/hooks'
import { loadTexture, Input, shapes } from 'fraxel'
import { createTrigger } from 'fraxel/hooks'

const PLAYER_TEX = await loadTexture('/assets/player.png')
const BULLET_TEX = await loadTexture('/assets/bullet.png')
const ENEMY_TEX = await loadTexture('/assets/enemy.png')

const fireTrigger = createTrigger<[x: number, y: number]>()
const Shoot = Input.createAction({ key: ' ' })

function Player() {
  const body = useTransform()
  const shoot = useAction(Shoot)

  useEffect(() => {
    if (shoot.justPressed()) {
      fireTrigger.emit(150, 200) // fire from player position
    }
  })

  return (
    <transform ref={body} position={[100, 200]}>
      <sprite textureId={PLAYER_TEX} sourceSize={[32, 48]} displaySize={[32, 48]} />
    </transform>
  )
}

function Bullet({ x, y }: { x: number; y: number }) {
  const body = useRigidBody()
  const collider = useCollider()

  useTrigger(collider.colliderEntered, () => {
    body.node.destroy()
  })

  return (
    <rigid-body ref={body} position={[x, y]} useGravity={false}>
      <sprite textureId={BULLET_TEX} sourceSize={[8, 4]} />
      <collider
        ref={collider}
        shape={shapes.circle(4)}
        group={['bullet']}
        collidesWith={['enemy']}
      />
    </rigid-body>
  )
}

function GameScene() {
  const container = useTransform()

  useTrigger(fireTrigger, (x, y) => {
    container.node.spawn(<Bullet x={x} y={y} />)
  })

  return (
    <transform ref={container}>
      <Player />
      <Enemy position={[350, 200]} />
    </transform>
  )
}
```
