The best way to understand how Fraxel works is to build something real. This guide walks you through creating a playable platformer — a character that runs, jumps, and lands on platforms with real physics. By the end you will have used the core building blocks: JSX scenes, body physics, input actions, and camera follow.

## Prerequisites

Make sure you have [installed Fraxel and configured TypeScript](/getting-started/installation) before continuing.

## Step 1 — Bootstrap the game

Create `src/main.tsx` with the minimal game setup. `createGame` takes a `<GameRoot>` JSX element and a DOM container, wires up the canvas, and returns game controls.

```tsx src/main.tsx
import { createGame, GameRoot, SceneRoot } from 'fraxel'

function Main() {
  return <transform />
}

const game = createGame(
  <GameRoot width={480} height={360} defaultScene='main'>
    <SceneRoot name='main' component={Main} />
  </GameRoot>,
  document.querySelector('#root')!,
)

game.play()
```

- **`<GameRoot width height defaultScene>`** — declares the canvas resolution and which scene to load first.
- **`<SceneRoot name component>`** — registers a named scene backed by a component function.
- **`createGame(jsx, root)`** — processes the JSX, creates the canvas inside `#root`, and returns game controls.
- **`game.play()`** — starts the render loop.

## Step 2 — Add platforms, a bouncing ball, and background

Static platforms are rigid bodies with `isStatic`. They don't move but still participate in collisions — other bodies bounce off them. We'll also add a bouncing ball and a dark background to give the scene depth.

```tsx src/main.tsx
import type { Shape, VectorLike } from 'fraxel'
import { createGame, GameRoot, SceneRoot } from 'fraxel'
import { shapes } from 'fraxel'

function Ball({ position }: { position: VectorLike }) {
  return (
    <body position={position} friction={0} bounce={1}>
      <geometry shape={shapes.circle(10)} fillColor={[0.2, 1, 0.3, 1]} />
      <collider
        shape={shapes.circle(10)}
        group={['ground']}
        collidesWith={['ground']}
      />
    </body>
  )
}

function Platform({ position, shape }: { position: VectorLike; shape: Shape }) {
  return (
    <body isStatic position={position} friction={0} bounce={1}>
      <geometry shape={shape} fillColor={[0.2, 0.3, 1, 1]} />
      <collider shape={shape} group={['ground']} collidesWith={['player']} />
    </body>
  )
}

function Main() {
  return (
    <transform>
      <geometry
        position={-250}
        shape={shapes.rectangle(1000, 1000)}
        fillColor={[0.05, 0.05, 0.09, 1]}
      />
      <Platform position={[-60, 460]} shape={shapes.rectangle(600, 20)} />
      <Platform position={[50, 380]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[300, 300]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[80, 220]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[320, 140]} shape={shapes.rectangle(100, 20)} />
      <Ball position={[10, 390]} />
    </transform>
  )
}

const game = createGame(
  <GameRoot width={480} height={360} defaultScene='main'>
    <SceneRoot name='main' component={Main} />
  </GameRoot>,
  document.querySelector('#root')!,
)

game.play()
```

Here's what's new:

- **`Shape` and `VectorLike`** — imported types from `fraxel` that let you pass any shape or position format to components.
- **`Platform` takes a `shape` prop** — instead of a fixed `width`, each platform receives its own `Shape`. This makes the component reusable for any shape.
- **`friction={0} bounce={1}`** — zero friction and full bounce make platforms and the ball perfectly elastic.
- **`Ball`** — a small green circle that bounces around the scene. It collides with other `"ground"` group bodies.
- **Background `<geometry>`** — a large dark rectangle positioned behind everything to fill the canvas with a near-black color.

## Step 3 — Add a player with physics

A `<body>` node adds gravity, velocity, and collision response. It **must** contain a `<collider>` child that defines the physical shape.

```tsx src/main.tsx
import type { Shape, VectorLike } from 'fraxel'
import { createGame, GameRoot, SceneRoot } from 'fraxel'
import { shapes } from 'fraxel'
import { useRigidBody } from 'fraxel'

function Ball({ position }: { position: VectorLike }) {
  return (
    <body position={position} friction={0} bounce={1}>
      <geometry shape={shapes.circle(10)} fillColor={[0.2, 1, 0.3, 1]} />
      <collider
        shape={shapes.circle(10)}
        group={['ground']}
        collidesWith={['ground']}
      />
    </body>
  )
}

function Platform({ position, shape }: { position: VectorLike; shape: Shape }) {
  return (
    <body isStatic position={position} friction={0} bounce={1}>
      <geometry shape={shape} fillColor={[0.2, 0.3, 1, 1]} />
      <collider shape={shape} group={['ground']} collidesWith={['player']} />
    </body>
  )
}

function Player() {
  const body = useRigidBody()

  return (
    <body ref={body} position={[220, 0]} mass={2}>
      <geometry shape={shapes.capsule(60, 20)} fillColor={[1, 0.1, 0.4, 1]} />
      <collider
        shape={shapes.capsule(60, 20)}
        group={['player']}
        collidesWith={['ground']}
      />
    </body>
  )
}

function Main() {
  return (
    <transform>
      <geometry
        position={-250}
        shape={shapes.rectangle(1000, 1000)}
        fillColor={[0.05, 0.05, 0.09, 1]}
      />
      <Platform position={[-60, 460]} shape={shapes.rectangle(600, 20)} />
      <Platform position={[50, 380]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[300, 300]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[80, 220]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[320, 140]} shape={shapes.rectangle(100, 20)} />
      <Ball position={[10, 390]} />
      <Player />
    </transform>
  )
}

const game = createGame(
  <GameRoot width={480} height={360} defaultScene='main'>
    <SceneRoot name='main' component={Main} />
  </GameRoot>,
  document.querySelector('#root')!,
)

game.play()
```

The player spawns at `[220, 0]` and immediately falls under gravity. When it hits a platform, the physics system resolves the collision — no manual collision detection needed.

## Step 4 — Add input handling

Fraxel's input system maps keys to semantic actions. `useAction` gives you reactive pressed state, and `useActionAxis` returns `-1`, `0`, or `1` based on two opposing keys.

```tsx src/main.tsx
import type { Shape, VectorLike } from 'fraxel'
import { createGame, GameRoot, SceneRoot } from 'fraxel'
import { Input, shapes } from 'fraxel'
import {
  useRigidBody,
  useAction,
  useActionAxis,
  useEffect,
  useUpdate,
} from 'fraxel'

const Left = Input.createAction({ key: 'a' })
const Right = Input.createAction({ key: 'd' })
const Jump = Input.createAction({ key: ' ' })

function Ball({ position }: { position: VectorLike }) {
  return (
    <body position={position} friction={0} bounce={1}>
      <geometry shape={shapes.circle(10)} fillColor={[0.2, 1, 0.3, 1]} />
      <collider
        shape={shapes.circle(10)}
        group={['ground']}
        collidesWith={['ground']}
      />
    </body>
  )
}

function Platform({ position, shape }: { position: VectorLike; shape: Shape }) {
  return (
    <body isStatic position={position} friction={0} bounce={1}>
      <geometry shape={shape} fillColor={[0.2, 0.3, 1, 1]} />
      <collider shape={shape} group={['ground']} collidesWith={['player']} />
    </body>
  )
}

function Player() {
  const body = useRigidBody()
  const direction = useActionAxis(Left, Right)
  const jump = useAction(Jump)

  useEffect(() => {
    if (!jump.pressed()) return
    body.applyImpulse([0, -400])
  })

  useUpdate(() => {
    body.setVelocity([direction() * 240, body.velocity().y])
  })

  return (
    <body ref={body} position={[220, 0]} mass={2}>
      <geometry shape={shapes.capsule(60, 20)} fillColor={[1, 0.1, 0.4, 1]} />
      <collider
        shape={shapes.capsule(60, 20)}
        group={['player']}
        collidesWith={['ground']}
      />
    </body>
  )
}

function Main() {
  return (
    <transform>
      <geometry
        position={-250}
        shape={shapes.rectangle(1000, 1000)}
        fillColor={[0.05, 0.05, 0.09, 1]}
      />
      <Platform position={[-60, 460]} shape={shapes.rectangle(600, 20)} />
      <Platform position={[50, 380]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[300, 300]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[80, 220]} shape={shapes.rectangle(100, 20)} />
      <Platform position={[320, 140]} shape={shapes.rectangle(100, 20)} />
      <Ball position={[10, 390]} />
      <Player />
    </transform>
  )
}

const game = createGame(
  <GameRoot width={480} height={360} defaultScene='main'>
    <SceneRoot name='main' component={Main} />
  </GameRoot>,
  document.querySelector('#root')!,
)

game.play()
```

Here's what's happening:

- **`Input.createAction({ key: 'a' })`** — creates a reusable action symbol bound to a key.
- **`useActionAxis(Left, Right)`** — returns `-1` when moving left, `1` when right, `0` when neither.
- **`useAction(Jump)`** — gives you `pressed()` which is `true` as long as the key is held.
- **`body.applyImpulse([0, -400])`** — instant upward velocity for jumping.
- **`body.setVelocity([x, y])`** — overrides velocity each frame for horizontal movement.
- **Early return pattern** — the effect bails out immediately if the jump key isn't pressed, keeping the logic flat.

## Step 5 — Add camera follow

The `<camera>` node follows a target and applies viewport transforms. Attach it to the player so the view tracks their movement.

```tsx src/main.tsx
import type { Shape, VectorLike } from 'fraxel'
import { createGame, GameRoot, SceneRoot } from 'fraxel'
import { Input, shapes } from 'fraxel'
import {
  useRigidBody,
  useAction,
  useActionAxis,
  useEffect,
  useUpdate,
} from 'fraxel'

const Left = Input.createAction({ key: 'a' })
const Right = Input.createAction({ key: 'd' })
const Jump = Input.createAction({ key: ' ' })

  return (
    <body position={position} friction={0} bounce={1}>
      <geometry shape={shapes.circle(10)} fillColor={[0.2, 1, 0.3, 1]} />
      <collider
        shape={shapes.circle(10)}
        group={['ground']}
        collidesWith={['ground']}
      />
    </body>
  )
}

function Platform({ position, shape }: { position: VectorLike; shape: Shape }) {
  return (
    <body isStatic position={position} friction={0} bounce={1}>
      <geometry shape={shape} fillColor={[0.2, 0.3, 1, 1]} />

}

      <Platform position={[80, 220]} shape={shapes.rectangle(100, 20)} />

    <SceneRoot name='main' component={Main} />
  </GameRoot>,
  document.querySelector('#root')!,
)

game.play()
```

- **`<camera current>`** — makes this the active camera that controls the viewport.
- **`smoothing={5}`** — camera interpolation speed (higher = snappier follow).
- **`offset={[-20, -30]}`** — shifts the camera so the player appears slightly left and above center.

## What you learned

In under 100 lines you built a platformer with:

- **Physics** — gravity, collision response, bounce, and jumping via `applyImpulse`
- **Input** — semantic actions with `Input.createAction` and reactive state with `useAction`
- **Platforms** — static rigid bodies that the player lands on, with configurable `Shape`
- **Ball** — a dynamic body with `bounce={1}` and `friction={0}` for perfect elasticity
- **Camera** — smooth follow attached to the player
- **Geometry** — colored shapes rendered without textures, including a dark background
- **Types** — `Shape` and `VectorLike` for flexible, reusable components
